# git 과 local folder 연결

		git init
		git remote add origin "repository address"


* * *

# git pull & push 방법

			pull: git pull origin master
			push: git status
				git add .
				git commit -m "message"
				git push origin +master


* * *

# git CRLF 에러
		
			git config --global core.autocrlf true
